1. Mapping a ShEX

Example notes on mapping a ShEX schema to a Wikidata data model from the IDEA project.


                    PREFIX p: <http://www.wikidata.org/prop/>
                    PREFIX ps: <http://www.wikidata.org/prop/statement/>
                    PREFIX pq: <http://www.wikidata.org/prop/qualifier/>
                    PREFIX psn: <http://www.wikidata.org/prop/statement/value-normalized/>
                    PREFIX wd: <http://www.wikidata.org/entity/>
                    PREFIX wdt: <http://www.wikidata.org/prop/direct/>
                    PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
                    PREFIX prov: <http://www.w3.org/ns/prov#>
                    PREFIX psv: <http://www.wikidata.org/prop/statement/value/>
                    PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
                    PREFIX wikibase: <http://wikiba.se/ontology#>
                    # QUERY MAP: SELECT ?item WHERE {?item wdt:P31 wd:Q220659.  ?item wdt:P5008 wd:Q114241199.}

                    start = @<idea_artifact_inscription>

                    <idea_artifact_inscription> {
    
                       # instance of
                        p:P31 {        ps:P31 [ wd:Q220659 ]    } + ;
                        # genre
                        p:P136 {        ps:P136 [ <http://www.wikidata.org/entity>~ ]} + ;
                        # image
                        p:P18 {        ps:P18 [ <http://commons.wikimedia.org/wiki/Special:FilePath>~ ] ;    } + ;
                        # creator
                        p:P170 {        ps:P170 [ <http://www.wikidata.org/.well-known/genid>~ ]} ; 
                        # depicts
                        p:P180  {      ps:P180 [ <http://www.wikidata.org/entity>~ ]   } + ;
                        # made from material
                        p:P186 {        ps:P186 [ <http://www.wikidata.org/entity>~ ]    } + ;
                        # location of discovery
                        p:P189 {        ps:P189 [ wd:Q464266 ]    } + ;
                        # collection
                        p:P195 {        ps:P195 [ wd:Q1568434 ]    } ;
                        # inventory number
                        p:P217 {        ps:P217 xsd:string    } ;
                        # location
                        p:P276 {        ps:P276 [ wd:Q49145]    } ;
                        # inception
                        p:P571 {        ps:P571 [  ] ;
                            psv:P571 [ <http://www.wikidata.org/value>~ ] ?} + ; 
                        # inscription   
                        p:P1684 {        ps:P1684 rdf:langString ;    } + ;
                        # height
                        p:P2048 {
                            psv:P2048 [ <http://www.wikidata.org/value>~ ] ;
                            # height
                            psn:P2048 [ <http://www.wikidata.org/value>~ ] ;
                            # height
                            ps:P2048 xsd:decimal
                        } ;
                        # width
                        p:P2049 {
                            # width
                            psv:P2049 [ <http://www.wikidata.org/value>~ ] ;
                            # width
                            psn:P2049 [ <http://www.wikidata.org/value>~ ] ;
                            # width
                            ps:P2049 xsd:decimal
                        } ;
                        # horizontal depth
                        p:P5524 {
                            psv:P5524 [ <http://www.wikidata.org/value>~ ] ;
                            # horizontal depth
                            psn:P5524 [ <http://www.wikidata.org/value>~ ] ;
                            # horizontal depth
                            ps:P5524 xsd:decimal
                        } ;
                        # copyright status
                        p:P6216 {        ps:P6216 [ wd:Q19652 ]    } ;
                        # Yale University Art Gallery ID
                        p:P8583 {        ps:P8583 xsd:string    } ;
                        # has part(s)
                        p:P527 {        ps:P527 [ <http://www.wikidata.org/entity>~ ]    } ? ;
                        # language of work or name
                        p:P407 {        ps:P407 [ <http://www.wikidata.org/entity>~ ]    } * ;
                        # described at URL
                        p:P973 {        ps:P973 [ <https://db.edcs.eu/epigr>~ ]    } ? ;
                        # described by source
                        p:P1343 {        ps:P1343 [ <http://www.wikidata.org/entity>~ ]    } ? ;
                        # dedicated to
                        p:P825 {        ps:P825 [ <http://www.wikidata.org/entity>~ ]    } ? ;
                        # title
                        p:P1476 {        ps:P1476 rdf:langString } ? ;
                        # mass
                        p:P2067 {
                            a [ wikibase:BestRank ] ;
                            # mass
                            psv:P2067 [ <http://www.wikidata.org/value>~ ] ;
                            # mass
                            psn:P2067 [ <http://www.wikidata.org/value>~ ] ;
                            # mass
                            ps:P2067 xsd:decimal
                        } ?
                    }
