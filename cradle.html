1. Mapping a ShEX

Example notes on mapping a ShEX schema to a Wikidata data model from the IDEA project.


                    PREFIX p: 
                    PREFIX ps: 
                    PREFIX pq: 
                    PREFIX psn: 
                    PREFIX wd: 
                    PREFIX wdt: 
                    PREFIX xsd: 
                    PREFIX prov: 
                    PREFIX psv: 
                    PREFIX rdf: 
                    PREFIX wikibase: 
                    # QUERY MAP: SELECT ?item WHERE {?item wdt:P31 wd:Q220659.  ?item wdt:P5008 wd:Q114241199.}

                    start = @

                     {
    
                       # instance of
                        p:P31 {        ps:P31 [ wd:Q220659 ]    } + ;
                        # genre
                        p:P136 {        ps:P136 [ ~ ]} + ;
                        # image
                        p:P18 {        ps:P18 [ ~ ] ;    } + ;
                        # creator
                        p:P170 {        ps:P170 [ ~ ]} ; 
                        # depicts
                        p:P180  {      ps:P180 [ ~ ]   } + ;
                        # made from material
                        p:P186 {        ps:P186 [ ~ ]    } + ;
                        # location of discovery
                        p:P189 {        ps:P189 [ wd:Q464266 ]    } + ;
                        # collection
                        p:P195 {        ps:P195 [ wd:Q1568434 ]    } ;
                        # inventory number
                        p:P217 {        ps:P217 xsd:string    } ;
                        # location
                        p:P276 {        ps:P276 [ wd:Q49145]    } ;
                        # inception
                        p:P571 {        ps:P571 [  ] ;
                            psv:P571 [ ~ ] ?} + ; 
                        # inscription   
                        p:P1684 {        ps:P1684 rdf:langString ;    } + ;
                        # height
                        p:P2048 {
                            psv:P2048 [ ~ ] ;
                            # height
                            psn:P2048 [ ~ ] ;
                            # height
                            ps:P2048 xsd:decimal
                        } ;
                        # width
                        p:P2049 {
                            # width
                            psv:P2049 [ ~ ] ;
                            # width
                            psn:P2049 [ ~ ] ;
                            # width
                            ps:P2049 xsd:decimal
                        } ;
                        # horizontal depth
                        p:P5524 {
                            psv:P5524 [ ~ ] ;
                            # horizontal depth
                            psn:P5524 [ ~ ] ;
                            # horizontal depth
                            ps:P5524 xsd:decimal
                        } ;
                        # copyright status
                        p:P6216 {        ps:P6216 [ wd:Q19652 ]    } ;
                        # Yale University Art Gallery ID
                        p:P8583 {        ps:P8583 xsd:string    } ;
                        # has part(s)
                        p:P527 {        ps:P527 [ ~ ]    } ? ;
                        # language of work or name
                        p:P407 {        ps:P407 [ ~ ]    } * ;
                        # described at URL
                        p:P973 {        ps:P973 [ ~ ]    } ? ;
                        # described by source
                        p:P1343 {        ps:P1343 [ ~ ]    } ? ;
                        # dedicated to
                        p:P825 {        ps:P825 [ ~ ]    } ? ;
                        # title
                        p:P1476 {        ps:P1476 rdf:langString } ? ;
                        # mass
                        p:P2067 {
                            a [ wikibase:BestRank ] ;
                            # mass
                            psv:P2067 [ ~ ] ;
                            # mass
                            psn:P2067 [ ~ ] ;
                            # mass
                            ps:P2067 xsd:decimal
                        } ?
                    }
